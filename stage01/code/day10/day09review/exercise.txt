class Printer：
    def __init__(self,brand,type):
        self.brand=brand
        self.type=type

    def printing():
        print("%s的%s正在打印。"%(self.brand,self.type))

class National_team:
    def __init__(self,nation,continent)
        self.nation=nation
        self.continent = continent

    def participate_world_cup():
        self.print("%s的%s队打进了世界杯。"%(self.continent,self.nation))
